Google Map Styler Sample Demo

Setup

yourhtml.html


        <!-- include googlemap main script -->
        <script src="http://maps.google.com/maps/api/js?key=INSERT_YOUR_GOOGLEMAP_API_KEY"></script>
        <!-- include plugin script -->
        <script src="INSERT_PATH_TO_YOUR_JS_FOLDER/js/googlemapstyler.js"></script>
        <!-- trigger plugin function -->
        <script>
            window.onload=function(){
                for (var i = document.getElementsByClassName("gmap").length - 1; i >= 0; i--) {
                    var _map = document.getElementsByClassName("gmap")[i];
                    var Gmap=new gmapStyler(_map,{});
                }
            }
        </script>
        </body>

yourcss.css


        /*define default map size*/
        .gmap,.gmap-inner{
            width:INSERT_MAP_WIDTH;
            height: INSERT_MAP_HEIGHT;
        }
